    **Inline JS** — Add at the bottom of sections/main-collection.liquid inside `<script type="module">`. This is the MainCollection web component extending ShopifySection.

    The JS covers four concerns:

    **1. Mobile filter drawer open/close:**
    - `filterToggleBtn` click: `mobileDrawer.setAttribute('aria-hidden', 'false')`, add `is-visible` to overlay, set `body.style.overflow = 'hidden'`, focus close button
    - Close btn click, overlay click: reverse — `aria-hidden="true"`, remove overlay class, restore body overflow, return focus to toggle button
    - Escape key: same as close (add handler in `onSectionLoad`, remove in `onSectionUnload`)

    **2. AJAX filter fetch (Section Rendering API):**
    - Listen to `change` on all `.filter-checkbox` elements
    - On change: build URL from all currently checked boxes, call `window.history.replaceState`, then fetch `${path}?${params}&sections=main-collection`
    - Parse JSON response: `data['main-collection']` is HTML string — use DOMParser to extract `.collection-grid` innerHTML, replace the live grid's innerHTML
    - After replacing grid HTML, re-bind Quick Add event listeners (new DOM nodes have no listeners)
    - Use `AbortController` to cancel in-flight request when new filter fires before previous resolves
    - Set `aria-busy="true"` on grid during fetch, remove after

    **3. Price range slider:**
    - For each `[data-price-slider]`: find min/max thumbs and fill bar
    - Update fill position: `left = (minVal - rangeMin) / (rangeMax - rangeMin) * 100%`, `width = maxPct - minPct`
    - Prevent thumb crossing: clamp min thumb below max, max thumb above min
    - Debounce `change` event (300ms) before applying filter fetch

    **4. Quick Add:**
    - Bind click handlers to all `.product-card__quick-add` buttons
    - `data-variants-count === "1"`: fetch `/products/{id}.js`, get `variants[0].id`, call `addToCart(variantId)`
    - `data-variants-count > 1`: fetch `/products/{id}.js`, render variant chip buttons in `.variant-popover__chips`, show popover positioned near trigger button (fixed positioning — viewport-aware: above if insufficient space below)
    - Chip click: select chip, enable add button, store `_selectedVariantId`
    - Add button click: call `addToCart(_selectedVariantId)`, close popover
    - Click outside popover: close popover (document click listener, check `!popover.contains(event.target)`)
    - `addToCart(variantId)`: POST to `/cart/add.js`, then GET `/cart.js` for total, publish `cart:updated` with `itemCount`, publish `cart:open`

    Full JS class structure:
    ```js
    class MainCollection extends window.BarterBobs.ShopifySection {
      onSectionLoad() {
        this._abortController = null;
        this._selectedVariantId = null;

        // Cache DOM refs
        this.mobileDrawer = this.querySelector('.mobile-filter-drawer');
        this.filterOverlay = this.querySelector('.filter-overlay');
        this.filterToggleBtn = this.querySelector('.filter-toggle-btn');
        this.grid = this.querySelector('.collection-grid');
        this.popover = this.querySelector('.variant-popover');
        this.popoverChips = this.querySelector('#variant-popover-chips');
        this.popoverAddBtn = this.querySelector('.variant-popover__add');

        // Mobile drawer
        this.filterToggleBtn?.addEventListener('click', () => this.openMobileFilter());
        this.querySelector('.mobile-filter-drawer__close')?.addEventListener('click', () => this.closeMobileFilter());
        this.filterOverlay?.addEventListener('click', () => this.closeMobileFilter());
        this.querySelector('.mobile-filter-apply')?.addEventListener('click', () => this.closeMobileFilter());
        this._escHandler = (e) => { if (e.key === 'Escape') this.closeMobileFilter(); };
        document.addEventListener('keydown', this._escHandler);

        // Filters
        this.querySelectorAll('.filter-checkbox').forEach(cb =>
          cb.addEventListener('change', () => this.applyFilter()));
        this.querySelectorAll('[data-price-slider]').forEach(s => this.initPriceSlider(s));

        // Quick Add
        this._bindQuickAddHandlers();
        this.popoverAddBtn?.addEventListener('click', () => this.addToCart(this._selectedVariantId));
        document.addEventListener('click', (e) => {
          if (this.popover && !this.popover.contains(e.target) &&
              !e.target.closest('.product-card__quick-add')) {
            this.closePopover();
          }
        });
      }

      onSectionUnload() {
        document.removeEventListener('keydown', this._escHandler);
      }

      _bindQuickAddHandlers() {
        this.querySelectorAll('.product-card__quick-add').forEach(btn =>
          btn.addEventListener('click', () => this.handleQuickAdd(btn)));
      }

      openMobileFilter() {
        this.mobileDrawer.setAttribute('aria-hidden', 'false');
        this.filterOverlay.classList.add('is-visible');
        this.filterToggleBtn.setAttribute('aria-expanded', 'true');
        document.body.style.overflow = 'hidden';
        this.querySelector('.mobile-filter-drawer__close')?.focus();
      }

      closeMobileFilter() {
        this.mobileDrawer.setAttribute('aria-hidden', 'true');
        this.filterOverlay.classList.remove('is-visible');
        this.filterToggleBtn.setAttribute('aria-expanded', 'false');
        document.body.style.overflow = '';
        this.filterToggleBtn.focus();
      }

      applyFilter() {
        const url = new URL(window.location.href);
        for (const key of [...url.searchParams.keys()]) {
          if (key.startsWith('filter.')) url.searchParams.delete(key);
        }
        this.querySelectorAll('.filter-checkbox:checked').forEach(cb =>
          url.searchParams.append(cb.name, cb.value));
        window.history.replaceState({}, '', url.toString());
        this.fetchFilteredGrid(url);
      }

      async fetchFilteredGrid(url) {
        if (this._abortController) this._abortController.abort();
        this._abortController = new AbortController();
        this.grid?.setAttribute('aria-busy', 'true');
        try {
          const res = await fetch(
            `${url.pathname}?${url.searchParams.toString()}&sections=main-collection`,
            { signal: this._abortController.signal }
          );
          const data = await res.json();
          const parsed = new DOMParser().parseFromString(data['main-collection'], 'text/html');
          const newGrid = parsed.querySelector('.collection-grid');
          if (this.grid && newGrid) {
            this.grid.innerHTML = newGrid.innerHTML;
            this._bindQuickAddHandlers();
          }
        } catch (err) {
          if (err.name !== 'AbortError') console.error('[MainCollection] Filter fetch failed:', err);
        } finally {
          this.grid?.removeAttribute('aria-busy');
          this._abortController = null;
        }
      }

      initPriceSlider(container) {
        const minThumb = container.querySelector('[data-thumb="min"]');
        const maxThumb = container.querySelector('[data-thumb="max"]');
        const fill = container.querySelector('[data-slider-fill]');
        const rangeMin = parseFloat(container.dataset.min);
        const rangeMax = parseFloat(container.dataset.max);
        const updateFill = () => {
          const lo = ((parseFloat(minThumb.value) - rangeMin) / (rangeMax - rangeMin)) * 100;
          const hi = ((parseFloat(maxThumb.value) - rangeMin) / (rangeMax - rangeMin)) * 100;
          fill.style.left = `${lo}%`;
          fill.style.width = `${hi - lo}%`;
        };
        minThumb.addEventListener('input', () => {
          if (parseFloat(minThumb.value) > parseFloat(maxThumb.value)) minThumb.value = maxThumb.value;
          updateFill();
        });
        maxThumb.addEventListener('input', () => {
          if (parseFloat(maxThumb.value) < parseFloat(minThumb.value)) maxThumb.value = minThumb.value;
          updateFill();
        });
        let debounce;
        [minThumb, maxThumb].forEach(t => t.addEventListener('change', () => {
          clearTimeout(debounce);
          debounce = setTimeout(() => {
            const url = new URL(window.location.href);
            url.searchParams.set(minThumb.name, minThumb.value);
            url.searchParams.set(maxThumb.name, maxThumb.value);
            window.history.replaceState({}, '', url.toString());
            this.fetchFilteredGrid(url);
          }, 300);
        }));
        updateFill();
      }

      async handleQuickAdd(btn) {
        const productId = btn.dataset.productId;
        const variantsCount = parseInt(btn.dataset.variantsCount, 10);
        if (variantsCount === 1) {
          btn.disabled = true;
          const prev = btn.textContent;
          btn.textContent = '...';
          try {
            const res = await fetch(`/products/${productId}.js`);
            const product = await res.json();
            await this.addToCart(product.variants[0].id);
          } finally {
            btn.disabled = false;
            btn.textContent = prev;
          }
        } else {
          await this.openVariantPopover(btn, productId);
        }
      }

      async openVariantPopover(triggerBtn, productId) {
        this._selectedVariantId = null;
        this.popoverAddBtn.disabled = true;
        const res = await fetch(`/products/${productId}.js`);
        const product = await res.json();
        this.popoverChips.innerHTML = product.variants
          .map(v => `<button class="variant-chip" data-variant-id="${v.id}"
            ${!v.available ? 'disabled' : ''}
            aria-label="${v.title}${!v.available ? ' (unavailable)' : ''}">
            ${v.title}</button>`).join('');
        this.popoverChips.querySelectorAll('.variant-chip:not([disabled])').forEach(chip => {
          chip.addEventListener('click', () => {
            this.popoverChips.querySelectorAll('.variant-chip').forEach(c => c.classList.remove('is-selected'));
            chip.classList.add('is-selected');
            this._selectedVariantId = chip.dataset.variantId;
            this.popoverAddBtn.disabled = false;
          });
        });
        const rect = triggerBtn.getBoundingClientRect();
        const spaceBelow = window.innerHeight - rect.bottom;
        this.popover.style.left = `${Math.max(8, rect.left)}px`;
        this.popover.style.width = `${Math.min(260, window.innerWidth - 16)}px`;
        if (spaceBelow < 180) {
          this.popover.style.top = '';
          this.popover.style.bottom = `${window.innerHeight - rect.top + 8}px`;
        } else {
          this.popover.style.bottom = '';
          this.popover.style.top = `${rect.bottom + 8}px`;
        }
        this.popover.setAttribute('aria-hidden', 'false');
        this.popoverChips.querySelector('.variant-chip:not([disabled])')?.focus();
      }

      closePopover() {
        this.popover?.setAttribute('aria-hidden', 'true');
        this._selectedVariantId = null;
      }

      async addToCart(variantId) {
        if (!variantId) return;
        try {
          const addRes = await fetch('/cart/add.js', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ id: variantId, quantity: 1 }),
          });
          if (!addRes.ok) throw new Error(`Cart add ${addRes.status}`);
          const cart = await fetch('/cart.js').then(r => r.json());
          window.BarterBobs.publish('cart:updated', { itemCount: cart.item_count });
          window.BarterBobs.publish('cart:open', {});
          this.closePopover();
        } catch (err) {
          console.error('[MainCollection] Quick Add failed:', err);
        }
      }
    }

    customElements.define('main-collection', MainCollection);
    ```
