Locked design decisions (from 01-CONTEXT.md):
  - Glassmorphism: background-color: var(--color-nav-bg) + -webkit-backdrop-filter: blur(12px) saturate(180%) + backdrop-filter: blur(12px) saturate(180%)
  - BOTH -webkit- prefix AND unprefixed form REQUIRED (iOS Safari)
  - transform: translateZ(0) on .site-header to create GPU compositing layer
  - Mobile nav drawer slides from LEFT (not right — cart drawer will come from right in Phase 3)
  - Mobile breakpoint: hamburger at < 768px; full horizontal nav at 768px+
  - Mobile nav overlay: semi-transparent dark backdrop; clicking overlay OR ✕ closes drawer
  - footer background: var(--color-surface-low) (#f2f3ff)
  - Footer is FULLY implemented in Phase 1 (not a placeholder)

Section group pattern (OS 2.0 — CRITICAL):
  sections/header-group.json: type must be "header" (not "header-group")
  sections/footer-group.json: type must be "footer" (not "footer-group")
  Both must have a "sections" object and "order" array

Anti-patterns to avoid:
  - {% include %} → use {% render %}
  - img_url filter → use image_url: width: N
  - window.* for state → use CustomEvent pub/sub
  - Hardcoded hex colors → use CSS custom properties

      // Handles hamburger toggle, overlay backdrop, Escape key, and drawer open/close

      class NavDrawer extends window.BarterBobs.ShopifySection {
        onSectionLoad() {
          this.hamburgerBtn = this.querySelector('[data-hamburger]');
          this.drawer = this.querySelector('[data-nav-drawer]');
          this.overlay = document.querySelector('[data-nav-overlay]');
          this.closeBtn = this.querySelector('[data-close-nav]');
          this._trapFocusCleanup = null;
          this._lastFocusedEl = null;

          this.hamburgerBtn?.addEventListener('click', this.open.bind(this));
          this.closeBtn?.addEventListener('click', this.close.bind(this));
          this.overlay?.addEventListener('click', this.close.bind(this));
          this._keydownHandler = this._handleKeydown.bind(this);
          document.addEventListener('keydown', this._keydownHandler);
        }

        onSectionUnload() {
          document.removeEventListener('keydown', this._keydownHandler);
          if (this._trapFocusCleanup) this._trapFocusCleanup();
        }

        open() {
          if (!this.drawer) return;
          this._lastFocusedEl = document.activeElement;
          this.drawer.setAttribute('aria-hidden', 'false');
          this.hamburgerBtn?.setAttribute('aria-expanded', 'true');
          this.overlay?.classList.add('is-visible');
          this.overlay?.setAttribute('aria-hidden', 'false');
          document.body.classList.add('nav-open');
          // Trap focus inside drawer — use trapFocus from global.js
          this._trapFocusCleanup = window.BarterBobs.trapFocus(this.drawer, this.closeBtn);
        }

        close() {
          if (!this.drawer) return;
          this.drawer.setAttribute('aria-hidden', 'true');
          this.hamburgerBtn?.setAttribute('aria-expanded', 'false');
          this.overlay?.classList.remove('is-visible');
          this.overlay?.setAttribute('aria-hidden', 'true');
          document.body.classList.remove('nav-open');
          // Release focus trap
          if (this._trapFocusCleanup) {
            this._trapFocusCleanup();
            this._trapFocusCleanup = null;
          }
          // Return focus to element that triggered the drawer
          this._lastFocusedEl?.focus();
          // Notify other components that the nav drawer closed
          window.BarterBobs.publish('drawer:close', { drawer: 'nav' });
        }

        _handleKeydown(event) {
          if (event.key === 'Escape' && this.drawer?.getAttribute('aria-hidden') === 'false') {
            this.close();
          }
        }
      }

      customElements.define('nav-drawer', NavDrawer);
    </script>
